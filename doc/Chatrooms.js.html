Define Chatrooms schema

var Chatrooms = sequelize.define('chatrooms', {
	cid: {
		type: Sequelize.INTEGER.UNSIGNED,
		allowNull: false,
		unique: true,
		autoIncrement: true,
		primaryKey: true
	},
	members: {
		type: Sequelize.TEXT,
		defaultValue: '',
		allowNull: false
	}
});

module.exports = Chatrooms;
